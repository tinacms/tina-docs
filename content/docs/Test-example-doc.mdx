## Step 1: Setting Up Your Component

Let's start by creating a simple `UserCard` component that displays user information.

```jsx
// components/UserCard.jsx
import React, { useState } from 'react';
import styles from './UserCard.module.css';

const UserCard = ({ user }) => {
  const [isExpanded, setIsExpanded] = useState(false);

  return (
    <div className={styles.userCard}>
      <div className={styles.header}>
        <img 
          src={user.avatar} 
          alt={`${user.name}'s avatar`}
          className={styles.avatar}
        />
        <div className={styles.info}>
          <h3 className={styles.name}>{user.name}</h3>
          <p className={styles.role}>{user.role}</p>
        </div>
      </div>
      
      <button 
        className={styles.toggleButton}
        onClick={() => setIsExpanded(!isExpanded)}
      >
        {isExpanded ? 'Show Less' : 'Show More'}
      </button>
      
      {isExpanded && (
        <div className={styles.details}>
          <p><strong>Email:</strong> {user.email}</p>
          <p><strong>Department:</strong> {user.department}</p>
          <p><strong>Bio:</strong> {user.bio}</p>
        </div>
      )}
    </div>
  );
};

export default UserCard;
```

## Step 3: Using Your Component

Now let's see how to use your `UserCard` component in a parent component:

```jsx
// App.jsx
import React from 'react';
import UserCard from './components/UserCard';

const App = () => {
  const sampleUsers = [
    {
      id: 1,
      name: "Sarah Johnson",
      role: "Frontend Developer",
      email: "sarah.johnson@example.com",
      department: "Engineering",
      bio: "Passionate about creating beautiful, accessible user interfaces with modern web technologies.",
      avatar: "https://images.unsplash.com/photo-1494790108755-2616b612b786?w=150"
    },
    {
      id: 2,
      name: "Michael Chen",
      role: "UX Designer",
      email: "michael.chen@example.com",
      department: "Design",
      bio: "Focused on user-centered design and creating intuitive digital experiences.",
      avatar: "https://images.unsplash.com/photo-1507003211169-0a1dd7228f2d?w=150"
    }
  ];

  return (
    <div style={{ maxWidth: '600px', margin: '0 auto', padding: '20px' }}>
      <h1>Team Directory</h1>
      {sampleUsers.map(user => (
        <UserCard key={user.id} user={user} />
      ))}
    </div>
  );
};

export default App;
```

## Step 4: Adding Tests

Let's write a simple test for our component using Jest and React Testing Library:

```jsx
// __tests__/UserCard.test.jsx
import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import UserCard from '../components/UserCard';

const mockUser = {
  name: "John Doe",
  role: "Developer",
  email: "john@example.com",
  department: "Engineering",
  bio: "A passionate developer",
  avatar: "https://example.com/avatar.jpg"
};

describe('UserCard Component', () => {
  test('renders user information correctly', () => {
    render(<UserCard user={mockUser} />);
    
    expect(screen.getByText('John Doe')).toBeInTheDocument();
    expect(screen.getByText('Developer')).toBeInTheDocument();
    expect(screen.getByText('Show More')).toBeInTheDocument();
  });

  test('toggles expanded view when button is clicked', () => {
    render(<UserCard user={mockUser} />);
    
    const toggleButton = screen.getByText('Show More');
    
    // Initially, details should not be visible
    expect(screen.queryByText('john@example.com')).not.toBeInTheDocument();
    
    // Click to expand
    fireEvent.click(toggleButton);
    
    // Details should now be visible
    expect(screen.getByText('john@example.com')).toBeInTheDocument();
    expect(screen.getByText('Show Less')).toBeInTheDocument();
  });

  test('displays user avatar with correct alt text', () => {
    render(<UserCard user={mockUser} />);
    
    const avatar = screen.getByAltText("John Doe's avatar");
    expect(avatar).toHaveAttribute('src', mockUser.avatar);
  });
});
```
