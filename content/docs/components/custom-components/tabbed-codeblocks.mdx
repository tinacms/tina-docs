Such like:

<queryResponseTabs
  query="// JavaScript
  fetch('/api/data')
  .then(res => res.json())
  .then(data => console.log(data))
  "
  response="// TypeScript
  async function getData(): Promise<void> {
  const res = await fetch('/api/data')
  const data = await res.json()
  console.log(data)
  }
  "
  customQueryName="JavaScript"
  customResponseName="TypeScript"
/>
